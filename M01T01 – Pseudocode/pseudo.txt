# Task 1: Enter Positive Numbers and Find Largest
# Repeatedly asks user to enter numbers and stores the largest
initialize max_number to 0
loop:
    request user to enter a positive number
    store input in variable number
    if number is 0:
        exit loop
    if number > max_number:
        set max_number to number
print "The largest number entered is", max_number

# Task 3: Arithmetic Average of Integers
# Reads multiple integers and calculates average
initialize total to 0
initialize count to 0
loop:
    request user to enter a number (or type 'done' to finish)
    if input is 'done':
        exit loop
    convert input to integer
    add number to total
    increment count by 1
if count > 0:
    average = total / count
    print "The average is", average
else:
    print "No numbers were entered"
